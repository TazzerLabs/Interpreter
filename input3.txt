print "Python program to display the Fibonacci sequence up to n-th term using"
print "a recursive function"

def recur_fibo(n):
   if n <= 1:
       return n
   else:
       return(recur_fibo(n-1) + recur_fibo(n-2))

nterms = 15

print "Fibonacci sequence:"

for i in range(nterms):
    print recur_fibo(i)
